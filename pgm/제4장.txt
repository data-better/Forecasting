library(forecast)
set.seed(1)
m <- list(
  arma=c(1,1,0,1,4,1,1),
  model=list(
    phi=0.5,
    theta=c(-0.5,  0.0,  0.0,  -0.5, 0.25)
  ),
  sigma2=1,
  x=NA
)
sarimasim <- ts(simulate.Arima(m, nsim=nn+21, future=FALSE),start=1970, frequency=4)
